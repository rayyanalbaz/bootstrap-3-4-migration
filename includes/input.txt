	<div class="entry-content">
		<h1>Welcome</h1>
<p>Since 1968, the <strong>Faculty of Environmental Studies (FES)</strong> at York University has been at the forefront of cutting edge research, teaching and learning. As the first Environmental Studies program in Canada, we explore the theoretical and practical relationships between the natural, social, artistic and built environments. Offering three innovative programs – <a href="/apply/bes/">Bachelor in Environmental Studies (BES)</a>, <a href="/apply/mes/">Master in Environmental Studies (MES)</a> and <a href="/apply/phd/">Doctorate in Environmental Studies (PhD)</a> – our mandate is to provide unsurpassed opportunities for our students.</p>
<p>Our vision is to direct and combine the diverse energies, assets and activities of FES as the premier location for interdisciplinary, analytical and collaborative research, education and community action on critical and changing environmental issues.</p>

(function ($) {
    $(document).ready(function () {
        $('.slide-down-arrow').on('click', function (e) {
            e.preventDefault();

            var target = this.hash;
            var target2 = $(target);

            $('html, body').stop().animate({
                'scrollTop': target2.offset().top - 48
            }, 900, 'swing', function () {
            });
        });
    });

    $(window).load(function () {
        var backgroundImg = $('.slide-background-img').text().replace('http://', 'http://');
        var backgroundColor = $('.slide-background-color').text();
        var fontColor = $('.slide-font-color').text();
        $('.slide-wrapper').css('background-color', backgroundColor);
        if (backgroundImg != '') {
            $('.slide-wrapper').css('background-image', 'url(' + backgroundImg + ')');
        } else {
            $('.slide-overlay').css('display', 'none');
        }

        if (fontColor != '') {
            $('.slide-heading').css('color', fontColor);
            $('.slide-sub-heading').css('color', fontColor);
            $('.slide-call-to-action-link').css('color', fontColor);
            $('.slide-call-to-action').css('color', fontColor);
        }

        setSliderHeight();
        centerSliderContent();
      
        $('.slide-content').fadeIn();
        $('.slide-down-arrow').fadeIn();
    });

    $(window).resize(function () {
        setSliderHeight();
        centerSliderContent();
    });

    function setSliderHeight() {
        var backgroundHeight = 100 - ($('.slide-wrapper').offset().top / $(window).height() * 100);
        $('.slide-wrapper').css('height', backgroundHeight + 'vh');
        $('.slide-wrapper').css('width', '100wh');
    }

    function centerSliderContent() {
        var sliderHeigh = $('.slide-wrapper').height();
        var sliderContentHeigh = $('.slide-content').height();
        var sliderContentTop = (sliderHeigh / 2) - (sliderContentHeigh / 2);
        $('.slide-content').css('top', sliderContentTop);
    }

})(jQuery);
